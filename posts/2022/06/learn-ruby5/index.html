陣列#

# 建立陣列
list = %w(apple banana cat)

p list # ["apple", "banana", "cat"]

# 陣列操作
puts list[0] # apple
puts list[1] # banana
puts list[-1] # cat
puts list[-2] # banana

puts list.first # apple
puts list.last # cat

puts list.length # 3

list << 'elephant'
puts list.last # elephant

list.push('fox')
puts list.last # fox

更多請參考官方手冊

map#

list = [1, 2, 3, 4, 5]
p list.map { |x| x * 2} # [2, 4, 6, 8, 10]

p (1..5).map { |x| x * 2} # [2, 4, 6, 8, 10]

select#

同 js 的 filter

p (1..5).select { |x| x <= 2} # [1, 2]

reduce#

p (1..5).reduce { |sum, n|  sum + n} # 15

練習#

compact 排除 nil

sort 排序

uniq 取唯一

list = [5, 4, 3, 2, nil, 1]
p list.compact.sort.uniq # [1, 2, 3, 4, 5]

範圍#

p (1..5).to_a # [1, 2, 3, 4, 5]

p (1...5).to_a # [1, 2, 3, 4]

練習2#

.odd 取單數

.odd? 相當於 true if n.odd?

p (1..100).select { |x| x.odd? } # [ 1, 3, 5, 7 .... 99]

p (1..100).reduce { |sum, n| sum + n} # 5050
p (1..100).sum # 5050

p (1...100).to_a.shuffle.first(5) # 轉 array, 隨機排序, 取前五
p (1...100).to_a.sample(5) # 轉 array, 隨機抽樣 5
